## Two-User Household Model

- One Calendar per household (shared)
- Both users have equal edit rights
- Invitations link new users to existing calendar
- User A creates calendar + invites User B
- User B accepts → auto-linked to calendar

## Invitation Workflow

1. User A (authenticated) → Settings → invite B@email.com
2. System checks: is B already linked? No → create CalendarInvitation (pending)
3. User B logs in (OAuth with B@email.com)
4. System detects pending invitation → auto-accept (or show accept button)
5. User B.calendar_id = User A.calendar_id
6. Both users query /api/events → see same data

## Event Visibility

When User A creates event:
- SET event.calendar_id = User A.calendar_id
- User B queries with User B.calendar_id → same results

   ```python
   from sqlalchemy.orm import Session
   from sqlalchemy import and_
   from app.database.models import User, Calendar, CalendarInvitation
   
   class UserRepository:
       def __init__(self, db: Session):
           self.db = db
       
       def get_user_by_id(self, user_id: str):
           return self.db.query(User).filter(User.id == user_id).first()
       
       def get_user_by_email(self, email: str):
           return self.db.query(User).filter(User.email == email).first()
       
       def create_invitation(self, calendar_id: str, invited_email: str, inviter_user_id: str):
           """Create pending invitation"""
           invitation = CalendarInvitation(
               calendar_id=calendar_id,
               invited_email=invited_email.lower(),
               inviter_user_id=inviter_user_id,
               status="pending"
           )
           self.db.add(invitation)
           self.db.commit()
           return invitation
       
       def get_pending_invitations(self, email: str):
           """Get all pending invitations for email"""
           return self.db.query(CalendarInvitation).filter(
               and_(
                   CalendarInvitation.invited_email == email.lower(),
                   CalendarInvitation.status == "pending"
               )
           ).all()
       
       def accept_invitation(self, invitation_id: str, user_id: str):
           """Accept invitation and link user to calendar"""
           invitation = self.db.query(CalendarInvitation).filter(
               CalendarInvitation.id == invitation_id
           ).first()
           if not invitation:
               return None
           
           user = self.db.query(User).filter(User.id == user_id).first()
           if not user:
               return None
           
           # Link user to calendar
           user.calendar_id = invitation.calendar_id
           invitation.status = "accepted"
           
           self.db.merge(user)
           self.db.merge(invitation)
           self.db.commit()
           return user
       
       def get_household_members(self, user_id: str):
           """Get all users in the same household/calendar"""
           user = self.get_user_by_id(user_id)
           if not user or not user.calendar_id:
               return []
           
           return self.db.query(User).filter(User.calendar_id == user.calendar_id).all()
       
       def get_household_calendar(self, user_id: str):
           """Get the shared calendar for this user"""
           user = self.get_user_by_id(user_id)
           if not user or not user.calendar_id:
               return None
           
           return self.db.query(Calendar).filter(Calendar.id == user.calendar_id).first()
   ```

2. **Create app/users/service.py** — Business logic:
   ```python
   from app.users.repository import UserRepository
   from app.database.models import User
   from sqlalchemy.orm import Session
   
   class UserService:
       def __init__(self, db: Session):
           self.repo = UserRepository(db)
           self.db = db
       
       def invite_user(self, inviter_id: str, invited_email: str):
           """
           User A invites User B.
           If User B already exists and is linked, return error.
           If User B doesn't exist, create pending invitation.
           """
           inviter = self.repo.get_user_by_id(inviter_id)
           if not inviter or not inviter.calendar_id:
               raise ValueError("Inviter has no calendar")
           
           invited_user = self.repo.get_user_by_email(invited_email.lower())
           
           # If already linked to this calendar, error
           if invited_user and invited_user.calendar_id == inviter.calendar_id:
               raise ValueError(f"{invited_email} is already a member")
           
           # If linked to different calendar, error (can't share across households)
           if invited_user and invited_user.calendar_id:
               raise ValueError(f"{invited_email} is already in another household")
           
           # Create pending invitation
           return self.repo.create_invitation(
               calendar_id=inviter.calendar_id,
               invited_email=invited_email.lower(),
               inviter_user_id=inviter_id
           )
       
       def accept_household_invitation(self, user_id: str):
           """
           User B logs in and has pending invitations.
           Auto-accept the first pending one.
           """
           user = self.repo.get_user_by_id(user_id)
           if not user:
               raise ValueError("User not found")
           
           # Get pending invitations for this user's email
           pending = self.repo.get_pending_invitations(user.email)
           if not pending:
               return None  # No pending invitations
           
           # Accept the first (earliest) invitation
           invitation = pending[0]
           result = self.repo.accept_invitation(invitation.id, user_id)
           return result
       
       def get_household_info(self, user_id: str):
           """Get calendar and all members"""
           calendar = self.repo.get_household_calendar(user_id)
           members = self.repo.get_household_members(user_id)
           
           return {
               'calendar': calendar,
               'members': members,
               'member_count': len(members)
           }
   ```

3. **Create app/users/routes.py** — API endpoints:
   ```python
   from fastapi import APIRouter, Depends, HTTPException
   from sqlalchemy.orm import Session
   from app.database.database import get_db
   from app.auth.dependencies import get_current_user
   from app.users.service import UserService
   from pydantic import BaseModel, EmailStr
   
   router = APIRouter(prefix="/api/users", tags=["users"])
   
   class InviteRequest(BaseModel):
       email: EmailStr
   
   class InvitationResponse(BaseModel):
       message: str
       invited_email: str
   
   @router.get("/me")
   async def get_current_user_info(user = Depends(get_current_user)):
       """Get current user info"""
       return {
           'id': user.id,
           'email': user.email,
           'name': user.name,
           'calendar_id': user.calendar_id,
           'last_login': user.last_login
       }
   
   @router.get("/household")
   async def get_household_info(
       user = Depends(get_current_user),
       db: Session = Depends(get_db)
   ):
       """Get household members and calendar info"""
       service = UserService(db)
       try:
           info = service.get_household_info(user.id)
           return {
               'calendar_id': info['calendar'].id if info['calendar'] else None,
               'calendar_name': info['calendar'].name if info['calendar'] else None,
               'member_count': info['member_count'],
               'members': [
                   {'id': m.id, 'email': m.email, 'name': m.name}
                   for m in info['members']
               ]
           }
       except Exception as e:
           raise HTTPException(status_code=400, detail=str(e))
   
   @router.post("/invite")
   async def invite_household_member(
       request: InviteRequest,
       user = Depends(get_current_user),
       db: Session = Depends(get_db)
   ):
       """Invite a user to the household"""
       if request.email.lower() == user.email.lower():
           raise HTTPException(status_code=400, detail="Cannot invite yourself")
       
       service = UserService(db)
       try:
           invitation = service.invite_user(user.id, request.email)
           return InvitationResponse(
               message=f"Invitation sent to {request.email}",
               invited_email=request.email
           )
       except ValueError as e:
           raise HTTPException(status_code=400, detail=str(e))
   
   @router.post("/accept-invitation")
   async def accept_invitation(
       user = Depends(get_current_user),
       db: Session = Depends(get_db)
   ):
       """Accept pending household invitation (auto-called on login)"""
       service = UserService(db)
       try:
           result = service.accept_household_invitation(user.id)
           if result:
               return {
                   'message': 'Invitation accepted',
                   'calendar_id': result.calendar_id
               }
           else:
               return {'message': 'No pending invitations'}
       except Exception as e:
           raise HTTPException(status_code=400, detail=str(e))
   ```
