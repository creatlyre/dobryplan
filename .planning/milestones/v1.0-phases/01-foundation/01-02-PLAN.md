## OAuth2 Pitfall: Token Exhaustion

From research: Google limits 100 refresh tokens per user per OAuth client. Rule: **Reuse existing refresh token — do NOT request new token if valid one exists.**

Strategy:
1. Store access_token, refresh_token, token_expiry in users table
2. When token expires, check if refresh_token still valid
3. Use refresh_token to get new access_token + optionally new refresh_token
4. If Google returns `invalid_grant`, user must re-authenticate (token was manually revoked)

## Session Flow

User → Google consent → Callback with auth code → Exchange for tokens → Create JWT → Set httpOnly cookie → Redirect to calendar

## Token Storage

OAuth tokens must be encrypted before storage in database. Use `cryptography` library's Fernet class (app/auth/utils.py).

   ```python
   from cryptography.fernet import Fernet
   from config import Settings
   
   # In production, use environment variable for key
   def get_cipher():
       settings = Settings()
       key = settings.DB_ENCRYPTION_KEY.encode() if settings.DB_ENCRYPTION_KEY else None
       if not key or len(key) < 32:
           raise ValueError("DB_ENCRYPTION_KEY must be 32+ chars")
       # Derive Fernet key from password
       from cryptography.hazmat.primitives import hashes
       from cryptography.hazmat.primitives.kdf.pbkdf2 import PBKDF2
       from cryptography.hazmat.backends import default_backend
       import base64
       kdf = PBKDF2(hashes.SHA256(), b"salt", 100000, 32, default_backend())
       derived = base64.urlsafe_b64encode(kdf.derive(key))
       return Fernet(derived)
   
   def encrypt_token(token: str) -> str:
       cipher = get_cipher()
       return cipher.encrypt(token.encode()).decode()
   
   def decrypt_token(encrypted_token: str) -> str:
       cipher = get_cipher()
       return cipher.decrypt(encrypted_token.encode()).decode()
   ```

2. **Create app/auth/oauth.py** — Google OAuth2 flow:
   ```python
   from google_auth_oauthlib.flow import Flow
   from google.audit.transport.requests import Request
   import os
   from config import Settings
   
   def get_flow():
       settings = Settings()
       return Flow.from_client_secrets_file(
           filename='client_secret.json',  # Download from Google Cloud Console
           scopes=settings.GOOGLE_SCOPES,
           redirect_uri=settings.GOOGLE_REDIRECT_URI
       )
   
   def get_authorization_url():
       """Generate Google OAuth2 consent URL"""
       flow = get_flow()
       auth_url, state = flow.authorization_url(
           access_type='offline',
           include_granted_scopes='true'
       )
       return auth_url, state
   
   def exchange_code_for_token(code: str, state: str):
       """Exchange auth code for access and refresh tokens"""
       try:
           flow = get_flow()
           flow.fetch_token(code=code)
           credentials = flow.credentials
           
           return {
               'access_token': credentials.token,
               'refresh_token': credentials.refresh_token,
               'token_expiry': credentials.expiry,
               'user_info': {
                   'email': credentials.id_token.get('email'),
                   'name': credentials.id_token.get('name'),
                   'google_id': credentials.id_token.get('sub')
               }
           }
       except Exception as e:
           raise ValueError(f"OAuth2 token exchange failed: {str(e)}")
   
   def refresh_access_token(refresh_token_encrypted: str):
       """Use refresh token to get new access token"""
       from app.auth.utils import decrypt_token
       try:
           refresh_token = decrypt_token(refresh_token_encrypted)
           flow = get_flow()
           flow.oauth2session.refresh_token(
               token_uri=flow.oauth2session.token_endpoint,
               refresh_token=refresh_token
           )
           return flow.oauth2session.token
       except ValueError as e:
           if "invalid_grant" in str(e):
               return None  # Token revoked, user must re-auth
           raise
   ```

3. **Create app/auth/routes.py** — OAuth login endpoints:
   ```python
   from fastapi import APIRouter, HTTPException, Depends, Response, Request
   from fastapi.responses import RedirectResponse
   from sqlalchemy.orm import Session
   from app.database.database import get_db
   from app.database.models import User, Calendar
   from app.auth.oauth import get_authorization_url, exchange_code_for_token
   from app.auth.utils import encrypt_token
   from config import Settings
   import uuid, secrets, jwt
   from datetime import datetime, timedelta
   
   router = APIRouter(prefix="/auth", tags=["auth"])
   
   @router.get("/login")
   async def login():
       """Redirect user to Google OAuth2 consent screen"""
       auth_url, state = get_authorization_url()
       # In production, verify state on callback (CSRF protection)
       return RedirectResponse(url=auth_url)
   
   @router.get("/callback")
   async def oauth_callback(code: str, state: str, db: Session = Depends(get_db)):
       """Handle OAuth2 callback from Google"""
       if not code:
           raise HTTPException(status_code=400, detail="Missing auth code")
       
       try:
           # Exchange code for tokens
           tokens = exchange_code_for_token(code, state)
           user_info = tokens['user_info']
           
           # Get or create user
           user = db.query(User).filter(User.google_id == user_info['google_id']).first()
           if not user:
               user = User(
                   id=str(uuid.uuid4()),
                   email=user_info['email'],
                   name=user_info['name'],
                   google_id=user_info['google_id'],
                   google_access_token=encrypt_token(tokens['access_token']),
                   google_refresh_token=encrypt_token(tokens['refresh_token'] or ""),
                   google_token_expiry=tokens['token_expiry'],
                   last_login=datetime.utcnow()
               )
               # Auto-create calendar for first-time user
               calendar = Calendar(
                   id=str(uuid.uuid4()),
                   name=f"{user_info['name']}'s Calendar",
                   owner_user_id=user.id
               )
               user.calendar_id = calendar.id
               db.add(calendar)
               db.add(user)
           else:
               # Update OAuth tokens for existing user
               user.google_access_token = encrypt_token(tokens['access_token'])
               if tokens['refresh_token']:
                   user.google_refresh_token = encrypt_token(tokens['refresh_token'])
               user.google_token_expiry = tokens['token_expiry']
               user.last_login = datetime.utcnow()
               db.merge(user)
           
           db.commit()
           db.refresh(user)
           
           # Create JWT session token
           settings = Settings()
           jwt_payload = {
               'user_id': str(user.id),
               'email': user.email,
               'exp': datetime.utcnow() + timedelta(hours=settings.JWT_EXPIRY_HOURS)
           }
           jwt_token = jwt.encode(jwt_payload, settings.SECRET_KEY, algorithm=settings.JWT_ALGORITHM)
           
           # Set secure, httpOnly cookie
           response = RedirectResponse(url="/")
           response.set_cookie(
               key="session",
               value=jwt_token,
               httponly=True,
               secure=True,  # HTTPS only in production
               samesite="lax",
               max_age=settings.JWT_EXPIRY_HOURS * 3600
           )
           return response
           
       except Exception as e:
           raise HTTPException(status_code=500, detail=f"OAuth callback failed: {str(e)}")
   
   @router.post("/logout")
   async def logout(response: Response):
       """Clear session cookie"""
       response.delete_cookie("session")
       return {"message": "Logged out"}
   ```

Implement session validation that preserves login state across page reloads:

1. **Create app/auth/dependencies.py** — FastAPI dependency for authenticated routes:
   ```python
   from fastapi import Depends, HTTPException, Cookie
   from sqlalchemy.orm import Session
   from app.database.database import get_db
   from app.database.models import User
   from config import Settings
   import jwt
   from typing import Optional
   
   async def get_current_user(
       session: Optional[str] = Cookie(None),
       db: Session = Depends(get_db)
   ):
       """Extract and validate JWT from session cookie"""
       if not session:
           raise HTTPException(status_code=401, detail="Not authenticated")
       
       try:
           settings = Settings()
           payload = jwt.decode(session, settings.SECRET_KEY, algorithms=[settings.JWT_ALGORITHM])
           user_id = payload.get("user_id")
           if not user_id:
               raise HTTPException(status_code=401, detail="Invalid session")
           
           user = db.query(User).filter(User.id == user_id).first()
           if not user:
               raise HTTPException(status_code=401, detail="User not found")
           
           return user
       except jwt.ExpiredSignatureError:
           raise HTTPException(status_code=401, detail="Session expired")
       except jwt.InvalidTokenError:
           raise HTTPException(status_code=401, detail="Invalid session")
   ```

2. **Create app/middleware/auth_middleware.py** — Global session validation:
   ```python
   from starlette.middleware.base import BaseHTTPMiddleware
   from starlette.requests import Request
   from starlette.responses import RedirectResponse
   from config import Settings
   import jwt
   from datetime import datetime, timedelta
   
   class SessionValidationMiddleware(BaseHTTPMiddleware):
       async def dispatch(self, request: Request, call_next):
           # Skip auth check for public routes
           public_routes = ["/auth/login", "/auth/callback", "/"]
           if request.url.path in public_routes:
               return await call_next(request)
           
           # Check session cookie
           session_cookie = request.cookies.get("session")
           if not session_cookie:
               return RedirectResponse(url="/auth/login")
           
           try:
               settings = Settings()
               payload = jwt.decode(session_cookie, settings.SECRET_KEY, algorithms=[settings.JWT_ALGORITHM])
               # Middleware just validates; get_current_user dependency will fetch user
               request.state.user_id = payload.get("user_id")
           except jwt.ExpiredSignatureError:
               return RedirectResponse(url="/auth/login")
           except jwt.InvalidTokenError:
               return RedirectResponse(url="/auth/login")
           
           return await call_next(request)
   ```

3. **Add middleware to main.py**:

   app.add_middleware(SessionValidationMiddleware)
