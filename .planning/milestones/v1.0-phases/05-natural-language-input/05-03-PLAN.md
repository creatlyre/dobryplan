         why: 'You said two different dates (e.g., "March 15 or April 20")',
         fix: 'Pick one date or rephrase: "March 15 at 2pm"'
       },
       'no_title': {
         what: 'I didn\'t catch what the event is about',
         why: 'You said when but not what',
         fix: 'Add an event name: "dentist Thursday 2pm"'
       },
       'network_error': {
         what: 'Network error',
         why: 'Could not reach the server',
         fix: 'Check your connection and try again'
       }
     };

     const msg = errorMessages[errorType] || {
       what: details || 'Parse failed',
       why: 'Unknown error',
       fix: 'Try rephrasing more clearly'
     };

     const errorSummary = document.getElementById('qa-error-summary');
     const errorWhy = document.getElementById('qa-error-why');
     
     errorSummary.textContent = msg.what;
     errorWhy.innerHTML = `<strong>Why:</strong> ${msg.why}<br/><strong>Fix:</strong> ${msg.fix}`;

     // Live region announcement
     const liveRegion = document.getElementById('qa-live-region');
     liveRegion.setAttribute('role', 'alert');
     liveRegion.textContent = `Error: ${msg.what}. ${msg.fix}`;
   }

1. **Focus Management:**
   - Modal should trap focus (Tab cycles within modal, not to page behind)
   - Add focusable element tracking:
   ```javascript
   const qaFocusableElements = 'button, [href], input, select, textarea, [tabindex]:not([tabindex="-1"])';
   
   function updateQAFocusables() {
     const visible = document.querySelectorAll(`#quick-add-modal ${qaFocusableElements}:not(:hidden)`);
     return Array.from(visible);
   }

   document.addEventListener('keydown', (e) => {
     if (e.key === 'Tab' && !qaModal.classList.contains('hidden')) {
       const focusables = updateQAFocusables();
       if (focusables.length === 0) return;

       const activeIndex = focusables.indexOf(document.activeElement);
       if (e.shiftKey) {
         // Shift+Tab
         if (activeIndex <= 0) {
           focusables[focusables.length - 1].focus();
           e.preventDefault();
         }
       } else {
         // Tab
         if (activeIndex >= focusables.length - 1) {
           focusables[0].focus();
           e.preventDefault();
         }
       }
     }
   });
   ```
