  ```javascript
  // Quick Add modal flow
  const qaModal = document.getElementById('quick-add-modal');
  const qaTextInput = document.getElementById('qa-text-input');
  const qaParseBtn = document.getElementById('qa-parse-btn');
  const qaErrorInline = document.getElementById('qa-error-inline');
  const qaReviewPhase = document.getElementById('qa-review-phase');
  const qaLoadingPhase = document.getElementById('qa-loading');
  const qaTextEntryPhase = document.getElementById('qa-text-entry');

  // Open modal
  document.getElementById('qa-open-btn').addEventListener('click', () => {
    qaModal.classList.remove('hidden');
    qaTextInput.focus();
    resetQuickAddModal();
  });

  // Close buttons
  const closeButtons = ['qa-close-btn-text', 'qa-back-btn'];
  closeButtons.forEach(id => {
    const btn = document.getElementById(id);
    if (btn) {
      btn.addEventListener('click', () => {
        qaModal.classList.add('hidden');
        resetQuickAddModal();
      });
    }
  });

  // Escape closes modal (except during save)
  document.addEventListener('keydown', (e) => {
    if (e.key === 'Escape' && !qaModal.classList.contains('hidden')) {
      // Check if save is in progress (look for disabled buttons)
      if (!document.getElementById('qa-save-btn')?.disabled) {
        qaModal.classList.add('hidden');
        resetQuickAddModal();
      }
    }
  });

  // Parse button
  qaParseBtn.addEventListener('click', async () => {
    const text = qaTextInput.value.trim();
    if (!text) {
      showQAError('Please enter some text', ['dentist Thursday 2pm', 'meeting next Friday 10am']);
      return;
    }

    qaLoadingPhase.classList.remove('hidden');
    qaErrorInline.classList.add('hidden');
    qaTextEntryPhase.classList.add('hidden');

    try {
      const contextDate = new Date(currentYear, currentMonth - 1, currentDay).toISOString().split('T')[0];
      const response = await fetch('/api/events/parse', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ text, context_date: contextDate })
      });

      if (!response.ok) {
        const data = await response.json();
        showQAError(
          data.errors?.[0] || 'Parse failed',
          data.errors?.slice(1) || []
        );
        qaLoadingPhase.classList.add('hidden');
        qaTextEntryPhase.classList.remove('hidden');
        return;
      }

      const parsed = await response.json();
      populateQAReview(text, parsed);
      qaLoadingPhase.classList.add('hidden');
      qaReviewPhase.classList.remove('hidden');
    } catch (error) {
      showQAError('Network error during parse', ['Check your connection and try again']);
      qaLoadingPhase.classList.add('hidden');
      qaTextEntryPhase.classList.remove('hidden');
    }
  });

  // Save Event / Save & Add Another
  const saveEvent = async (keepOpen = false) => {
    const payload = buildQAPayload();
    if (!payload.title || !payload.start_at) {
      alert('Title and start date are required');
      return;
    }

    document.getElementById('qa-save-btn').disabled = true;
    document.getElementById('qa-save-another-btn').disabled = true;

    try {
      const response = await fetch('/api/events', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(payload)
      });

      if (response.ok) {
        showQAToast('Event saved');
        if (keepOpen) {
          resetQuickAddModal();
          qaTextInput.focus();
        } else {
          qaModal.classList.add('hidden');
          await refreshPanels();
        }
      } else {
        const err = await response.json();
        showQAError(err.detail || 'Save failed', []);
      }
    } finally {
      document.getElementById('qa-save-btn').disabled = false;
      document.getElementById('qa-save-another-btn').disabled = false;
    }
  };

  document.getElementById('qa-save-btn').addEventListener('click', async () => {
    await saveEvent(false);
  });

  document.getElementById('qa-save-another-btn').addEventListener('click', async () => {
    await saveEvent(true);
  });

  // Back button returns to text entry
  document.getElementById('qa-back-btn').addEventListener('click', () => {
    qaReviewPhase.classList.add('hidden');
    qaTextEntryPhase.classList.remove('hidden');
    qaTextInput.focus();
  });

  function populateQAReview(originalText, parsed) {
    document.getElementById('qa-review-original-text').textContent = originalText;
    document.getElementById('qa-parsed-title').value = parsed.title || '';
    
    // Convert ISO datetime to datetime-local format
    const startDateTime = new Date(parsed.start_at).toLocaleString('sv-SE', { year: 'numeric', month: '2-digit', day: '2-digit', hour: '2-digit', minute: '2-digit' }).replace(' ', 'T');
    document.getElementById('qa-parsed-start').value = startDateTime;
    
    if (parsed.end_at) {
      const endDateTime = new Date(parsed.end_at).toLocaleString('sv-SE', { year: 'numeric', month: '2-digit', day: '2-digit', hour: '2-digit', minute: '2-digit' }).replace(' ', 'T');
      document.getElementById('qa-parsed-end').value = endDateTime;
    }

    if (parsed.recurrence?.freq) {
      document.getElementById('qa-parsed-repeat').value = parsed.recurrence.freq;
      if (parsed.recurrence.count) document.getElementById('qa-parsed-count').value = parsed.recurrence.count;
      if (parsed.recurrence.until) document.getElementById('qa-parsed-until').value = parsed.recurrence.until.split('T')[0];
    }
  }

  function buildQAPayload() {
    return {
      title: document.getElementById('qa-parsed-title').value,
      start_at: new Date(document.getElementById('qa-parsed-start').value).toISOString(),
      end_at: new Date(document.getElementById('qa-parsed-end').value).toISOString(),
      timezone: Intl.DateTimeFormat().resolvedOptions().timeZone || 'UTC',
      rrule: buildRRule(
        document.getElementById('qa-parsed-repeat').value,
        document.getElementById('qa-parsed-count').value,
        document.getElementById('qa-parsed-until').value
      )
    };
  }

  function resetQuickAddModal() {
    qaTextEntryPhase.classList.remove('hidden');
    qaReviewPhase.classList.add('hidden');
    qaLoadingPhase.classList.add('hidden');
    qaErrorInline.classList.add('hidden');
    qaTextInput.value = '';
  }

  function showQAError(message, examples) {
    qaErrorInline.classList.remove('hidden');
    document.getElementById('qa-error-summary').textContent = message;
    if (examples.length > 0) {
      document.getElementById('qa-examples-list').textContent = examples.slice(0, 2).join(', ');
    }
  }

  function showQAToast(message) {
    // Use existing toast mechanism or create simple div
    const toast = document.createElement('div');
    toast.className = 'fixed bottom-4 right-4 bg-green-600 text-white px-4 py-2 rounded shadow-lg';
    toast.textContent = message;
    document.body.appendChild(toast);
    setTimeout(() => toast.remove(), 3000);
  }
  ```
